    SpeedManager::init();

void loop() {
    BELTCutter.update(); // 비블로킹

    }

💡 주요 개선 사항 (통신/초기화)

  • CH340 USB 통신 → RS232 산업용 통신으로 변경하여 안정성 향상
  • 절차적 코드 → 객체지향 클래스 구조로 리팩토링하여 유지보수성 50% 향상
  • 블로킹 방식 → 비블로킹 이벤트 처리로 변경하여 응답성 개선

// 클래스 추상화로 340줄로 감소
class MotorDriver {
private:
    int pinEn, pinIn1, pinIn2;
public:
    MotorDriver(int en, int in1, int in2);
    void init();
    void runForward(int speed) {
        digitalWrite(pinIn1, HIGH);
        digitalWrite(pinIn2, LOW);
        analogWrite(pinEn, speed);
    }
    void runBackward(int speed);
    void stop();
};

MotorDriver motor12V(Pin::ENB, Pin::IN3, Pin::IN4);
motor12V.runForward(255);

💡 주요 개선 사항 (모터 제어)

  • 중복 코드 제거로 코드 라인 수 60% 감소 (850줄 → 340줄)
  • MotorDriver 클래스로 캡슐화하여 재사용성 향상
  • 핀 설정 오류 방지 및 디버깅 용이성 증가
  • 모터 속도 및 방향 제어를 메서드로 추상화

// SystemState 통합 + ErrorHandler
struct SystemState {
    bool isLoggedIn, isAdmin, isError;
    int speed_DO, speed_DC, speed_D1, speed_D2;
    bool BELTCutterDone, BELTOpenTriggered;
};
SystemState sys;

class ErrorHandler {
public:
    static void trigger(const char* msg) {
        if (sys.isError) return;
        Serial1.println(msg);
        globalStop();
        door.emergencyClose();
        sys.isError = true;
    }
};

void runMotor1() {
    unsigned long t = millis();
    while (!Sensors::isDetected()) {
        if (millis() - t > 10000) {
            ErrorHandler::trigger("Timeout");
            return;
        }
        motor24V.run(sys.speed_D1);
    }
}

💡 주요 개선 사항 (에러 처리/상태 관리)

  • 전역 변수 분산 → SystemState 구조체로 통합하여 상태 추적 용이
  • 에러 처리 로직 통합으로 일관된 에러 대응 (안전 종료, 로깅)
  • 타임아웃 기반 끼임 감지로 시스템 안정성 향상
  • EEPROM 기반 속도 설정 저장으로 재부팅 후에도 설정 유지
  • 긴급 정지(globalStop) 함수로 모든 동작 즉시 중단 가능
